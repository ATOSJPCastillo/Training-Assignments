        {

3. Create a method to swap two numbers.

package mypack;
import java.util.Scanner;  
public class JavaAssignment3  {  

	public static void main(String args[])  {
	int x, y;  
	Scanner sc = new Scanner(System.in);  
	
	System.out.print("Enter first number: ");  
	x = sc.nextInt();  
	
	System.out.print("Enter second number: ");  
	y = sc.nextInt();  
	
	System.out.println("\nBefore Swapping:\nfirst = "+x+"\nsecond = "+y);  
	swap(x, y);  
	
	}  
	
	public static void swap(int a, int b)  
	{       
		a = a + b;  
		b = a - b;  
		a = a - b;  
		
		System.out.println("\nAfter Swapping:\nfirst = "+a+"\nsecond = "+b);  
	}  
}

package mypack;
import java.util.Scanner;
public class JavaAssignment4 {
	
  public static void main(String[] args) {

    char operator;
    int num1, num2, res;

    Scanner input = new Scanner(System.in);

    System.out.println("Enter the first number");
    num1 = input.nextInt();

    System.out.println("Enter the second number");
    num2 = input.nextInt();

    System.out.println("Choose operator: ( +, -, *, / )");
    operator = input.next().charAt(0);
    
    
    switch (operator) {

      case '+':
        res = num1 + num2;
        System.out.println(num1 + " + " + num2 + " = " + res);
        break;

      case '-':
        res = num1 - num2;
        System.out.println(num1 + " - " + num2 + " = " + res);
        break;

      case '*':
        res = num1 * num2;
        System.out.println(num1 + " * " + num2 + " = " + res);
        break;

      case '/':
        res = num1 / num2;
        System.out.println(num1 + " / " + num2 + " = " + res);
        break;

      default:
        System.out.println("Invalid operator!");
        break;
    }

    input.close();
  }
}
